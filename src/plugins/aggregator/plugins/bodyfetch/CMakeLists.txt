include_directories (${AGGREGATOR_INCLUDE_DIR}
	${CMAKE_CURRENT_BINARY_DIR}
	)

	bodyfetch.cpp
	workerobject.cpp
	)
set (BODYFETCH_HEADERS
	bodyfetch.h
	workerobject.h
	)

	${BODYFETCH_RCCS}

	)

	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
	)
