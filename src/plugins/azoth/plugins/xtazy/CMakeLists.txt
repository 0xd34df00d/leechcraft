INCLUDE_DIRECTORIES (${AZOTH_INCLUDE_DIR}
	${CMAKE_CURRENT_BINARY_DIR}
	)
SET (XTAZY_SRCS
	xtazy.cpp
	tunesourcebase.cpp
	filesource.cpp
	xmlsettingsmanager.cpp
	)
SET (XTAZY_HEADERS
	xtazy.h
	tunesourcebase.h
	filesource.h
	xmlsettingsmanager.h
	)
SET (XTAZY_RESOURCES azothxtazyresources.qrc)
CreateTrVars ("azoth_xtazy" "en;es;ru_RU;uk_UA" XTAZY_TRANSLATIONS XTAZY_COMPILED_TRANSLATIONS)
QT4_ADD_RESOURCES (XTAZY_RCCS ${XTAZY_RESOURCES})

IF (QT_QTDBUS_FOUND)
	ADD_DEFINITIONS (-DHAVE_DBUS)

	SET (XTAZY_SRCS
		${XTAZY_SRCS}
		mprissource.cpp
		)
	SET (XTAZY_HEADERS
		${XTAZY_HEADERS}
		mprissource.h
		)
ENDIF (QT_QTDBUS_FOUND)

IF (NOT LC_NO_MOC)
	QT4_WRAP_CPP (XTAZY_MOC_SRCS ${XTAZY_HEADERS})
ENDIF (NOT LC_NO_MOC)

ADD_CUSTOM_COMMAND (OUTPUT
	${XTAZY_COMPILED_TRANSLATIONS}
	COMMAND "${QT_LRELEASE_EXECUTABLE}" ${XTAZY_TRANSLATIONS}
	DEPENDS ${XTAZY_TRANSLATIONS}
	WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
	)
ADD_LIBRARY (leechcraft_azoth_xtazy SHARED
	${XTAZY_COMPILED_TRANSLATIONS}
	${XTAZY_SRCS}
	${XTAZY_MOC_SRCS}
	${XTAZY_RCCS}
	)
TARGET_LINK_LIBRARIES (leechcraft_azoth_xtazy
	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
	)
INSTALL (TARGETS leechcraft_azoth_xtazy DESTINATION ${LC_PLUGINS_DEST})
INSTALL (FILES azothxtazysettings.xml DESTINATION ${LC_SETTINGS_DEST})
INSTALL (FILES ${XTAZY_COMPILED_TRANSLATIONS} DESTINATION ${LC_TRANSLATIONS_DEST})
