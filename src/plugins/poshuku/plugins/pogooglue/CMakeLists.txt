INCLUDE_DIRECTORIES (${POSHUKU_INCLUDE_DIR}
	${CMAKE_CURRENT_BINARY_DIR})
SET (POGOOGLUE_SRCS
	pogooglue.cpp
	)
SET (POGOOGLUE_HEADERS
	pogooglue.h
	)
SET (POGOOGLUE_RESOURCES poshukupogooglueresources.qrc)
CreateTrVars ("poshuku_pogooglue" "en;ru_RU;uk_UA" POGOOGLUE_TRANSLATIONS POGOOGLUE_COMPILED_TRANSLATIONS)

IF (NOT LC_NO_MOC)
	QT4_WRAP_CPP (POGOOGLUE_MOC_SRCS ${POGOOGLUE_HEADERS})
ENDIF (NOT LC_NO_MOC)
QT4_ADD_RESOURCES (POGOOGLUE_RCCS ${POGOOGLUE_RESOURCES})

ADD_CUSTOM_COMMAND (OUTPUT
	${POGOOGLUE_COMPILED_TRANSLATIONS}
	COMMAND "${QT_LRELEASE_EXECUTABLE}" ${POGOOGLUE_TRANSLATIONS}
	DEPENDS ${POGOOGLUE_TRANSLATIONS}
	WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
	)
ADD_LIBRARY (leechcraft_poshuku_pogooglue SHARED
	${POGOOGLUE_COMPILED_TRANSLATIONS}
	${POGOOGLUE_SRCS}
	${POGOOGLUE_MOC_SRCS}
	${POGOOGLUE_RCCS}
	)
TARGET_LINK_LIBRARIES (leechcraft_poshuku_pogooglue
	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
	)
INSTALL (TARGETS leechcraft_poshuku_pogooglue DESTINATION ${LC_PLUGINS_DEST})
INSTALL (FILES ${POGOOGLUE_COMPILED_TRANSLATIONS} DESTINATION ${LC_TRANSLATIONS_DEST})
