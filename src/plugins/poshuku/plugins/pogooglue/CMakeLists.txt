INCLUDE_DIRECTORIES (${POSHUKU_INCLUDE_DIR}
	${CMAKE_CURRENT_BINARY_DIR})
SET (POGOOGLUE_SRCS
	pogooglue.cpp
	)
SET (POGOOGLUE_HEADERS
	pogooglue.h
	)

SET (POGOOGLUE_RESOURCES poshukupogooglueresources.qrc
	)

IF (NOT LC_NO_MOC)
	QT4_WRAP_CPP (POGOOGLUE_MOC_SRCS ${POGOOGLUE_HEADERS})
ENDIF (NOT LC_NO_MOC)
QT4_ADD_RESOURCES (POGOOGLUE_RCCS ${POGOOGLUE_RESOURCES})

ADD_LIBRARY (leechcraft_poshuku_pogooglue SHARED
	${POGOOGLUE_SRCS}
	${POGOOGLUE_MOC_SRCS}
	${POGOOGLUE_RCCS}
	)
TARGET_LINK_LIBRARIES (leechcraft_poshuku_pogooglue
	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
	)
INSTALL (TARGETS leechcraft_poshuku_pogooglue DESTINATION ${LC_PLUGINS_DEST})
